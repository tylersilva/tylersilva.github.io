import ipywidgets as widgets

widgets.IntSlider(
    min=0,
    max=10,
    step=1,
    description='Slider:',
    value=3
)

slider = widgets.IntSlider()
text = widgets.IntText()
display(slider, text)

widgets.jslink((slider, 'value'), (text, 'value'))

Link(source=(IntSlider(value=0), 'value'), target=(IntText(value=0), 'value'))
